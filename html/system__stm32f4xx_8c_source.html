203 void SystemCoreClockUpdate(void)
204 {
205  uint32_t tmp = 0, pllvco = 0, pllp = 2, pllsource = 0, pllm = 2;
206 
207  /* Get SYSCLK source -------------------------------------------------------*/
208  tmp = RCC->CFGR & RCC_CFGR_SWS;
209 
210  switch (tmp)
211  {
212  case 0x00: /* HSI used as system clock source */
213  SystemCoreClock = HSI_VALUE;
214  break;
215  case 0x04: /* HSE used as system clock source */
216  SystemCoreClock = HSE_VALUE;
217  break;
218  case 0x08: /* PLL used as system clock source */
219 
220  /* PLL_VCO = (HSE_VALUE or HSI_VALUE / PLL_M) * PLL_N
221  SYSCLK = PLL_VCO / PLL_P
222  */
223  pllsource = (RCC->PLLCFGR & RCC_PLLCFGR_PLLSRC) >> 22;
224  pllm = RCC->PLLCFGR & RCC_PLLCFGR_PLLM;
225 
226  if (pllsource != 0)
227  {
228  /* HSE used as PLL clock source */
229  pllvco = (HSE_VALUE / pllm) * ((RCC->PLLCFGR & RCC_PLLCFGR_PLLN) >> 6);
230  }
231  else
232  {
233  /* HSI used as PLL clock source */
234  pllvco = (HSI_VALUE / pllm) * ((RCC->PLLCFGR & RCC_PLLCFGR_PLLN) >> 6);
235  }
236 
237  pllp = (((RCC->PLLCFGR & RCC_PLLCFGR_PLLP) >>16) + 1 ) *2;
238  SystemCoreClock = pllvco/pllp;
239  break;
240  default:
241  SystemCoreClock = HSI_VALUE;
242  break;
243  }
244  /* Compute HCLK frequency --------------------------------------------------*/
245  /* Get HCLK prescaler */
246  tmp = AHBPrescTable[((RCC->CFGR & RCC_CFGR_HPRE) >> 4)];
247  /* HCLK frequency */
248  SystemCoreClock >>= tmp;
249 }
