334 void HAL_ADC_ConvCpltCallback(ADC_HandleTypeDef* hadc)
335 {
336  // GUARDA OS VALORES DOS SENSORES LDR E YL-69
337  if(hadc->Instance==ADC1) // Verifica se o CallBack foi feito pelo ADC1
338  { // Início da conversao completa do ADC1
339  average[LDR] = 0; average[YL_69] = 0; // Zera as posições dos vetors de média
340  for(int i=0;i<SAMPLE;i++) // Laço para guardar as medidas
341  { //
342  if((i%2)==0)average[LDR] += measure_ADC1[i]; // Os valores pares são o LDR, soma todas as medições e salva no vetor
343  else average[YL_69] += measure_ADC1[i]; // Os valores ímpares são o YL-69, soma todas as medições e salva no vetor
344  } //
345  Light = average[LDR]*PERCENT; // Retorna a porcentagem do valor medido de cada sensor
346  Moisture = average[YL_69]*PERCENT; // Retorna a porcentagem do valor medido de cada sensor
347  } // Fim da conversão completa do ADC1
348  // GUARDA OS VALORES DO SENSOR DE CORRENTE
349  if(hadc->Instance==ADC2) // Verifica se o CallBack foi feito pelo ADC2
350  { // Início da conversao completa do ADC2
351  average[ACS712] = 0; // Zera as posições do vetor de média
352  for(int i=0;i<SAMPLE;i++) // Laço para guardar as medidas
353  { //
354  average[ACS712] += measure_ADC2[i]; // Soma todas as medições e salva no vetor
355  } //
356  // CALCULO DA CORRENTE ELÉTRICA
357  average[ACS712] /= SAMPLE; // Calcula a média aritmética das medidas
358  Current = (((3122.0-average[ACS712])/4096.0)*(3.3/C_SENS)); // Calcula a corrente elétrica a partir da medição do sensor de corrente
359  Current = abs(Current*1000)/1000.0; // Arredonda o módulo do valor em 3 casas decimais
360  // CONTROLE DE SOBRECORRENTE
361  if(Current>I_MAX) // Se a corrente atual ultrapassa o valor máximo

364  } // Fim da condição de sobrecorrente
365  // CALCULO DA POTÊNCIA ELÉTRICA

367  } // Fim da conversão completa do ADC2
