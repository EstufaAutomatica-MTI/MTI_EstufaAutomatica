23 uint32_t DWT_Delay_Init(void)

26  CoreDebug->DEMCR &= ~CoreDebug_DEMCR_TRCENA_Msk; // ~0x01000000;

30  /* Disable clock cycle counter */

32  /* Enable clock cycle counter */
33  DWT->CTRL |= DWT_CTRL_CYCCNTENA_Msk; //0x00000001;
34 
35  /* Reset the clock cycle counter value */
36  DWT->CYCCNT = 0;
37 
38  /* 3 NO OPERATION instructions */
39  __ASM volatile ("NOP");
40  __ASM volatile ("NOP");
41  __ASM volatile ("NOP");

44  if(DWT->CYCCNT)

46  return 0; /*clock cycle counter started*/
47  }

58  /* Go to number of cycles for system */

118 uint8_t DHT_Read (void)
119 {
120  uint8_t i,j;
121  for (j=0;j<8;j++)
122  {
123  while (!(HAL_GPIO_ReadPin (DHT_PORT, DHT_PIN))); // wait for the pin to go high
124  delay (40); // wait for 40 us
125  if (!(HAL_GPIO_ReadPin (DHT_PORT, DHT_PIN))) // if the pin is low
126  {
127  i&= ~(1<<(7-j)); // write 0
128  }
129  else i|= (1<<(7-j)); // if the pin is high, write 1
130  while ((HAL_GPIO_ReadPin (DHT_PORT, DHT_PIN))); // wait for the pin to go low
131  }
132  return i;
133 }
